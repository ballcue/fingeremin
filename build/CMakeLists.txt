# assemble hist.asm
add_custom_command (
    OUTPUT hist.o
    COMMAND nasm -f elf64 -g -F dwarf ../asm/hist.asm -o hist.o
    DEPENDS ../asm/hist.asm
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
)

# assemble backproject.asm
add_custom_command (
    OUTPUT backproject.o

    DEPENDS ../asm/backproject.asm
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
)

# assemble meanshift.asm
